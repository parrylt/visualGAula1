Algoritmo "IMC"
//  
//  

Var
  peso, altura, imc : real
Inicio
  escreva("Digite seu peso: ")
  leia(peso)
  escreva("Digite sua altura: ")
  leia(altura)
  imc <- peso / (altura * altura)
  
  se imc < 18.5 entao
    escreva("Seu imc é de ", imc:6:2, " - Abaixo do peso")
  fimse
  se imc >= 18.5 entao
    se imc < 25 entao
      escreva("Seu imc é de ", imc:6:2, " - Peso normal")
    fimse
  fimse
  se imc >= 25 entao
    se imc < 30 entao
      escreva("Seu imc é de ", imc:6:2, " - Acima do peso")
    fimse
  fimse
  se imc >= 30 entao
    se imc < 35 entao
      escreva("Seu imc é de ", imc:6:2, " - Obesidade I")
    fimse
  fimse
  se imc >= 35 entao
    se imc < 40 entao
      escreva("Seu imc é de ", imc:6:2, " - Obesidade II")
    fimse
  fimse
  se imc >= 40 entao
    escreva("Seu imc é de ", imc:6:2, " - Obesidade III")
  fimse
fimalgoritmo
